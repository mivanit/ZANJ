 33def jsonl_metadata(data: list[JSONdict]) -> dict:
 34    """metadata about a jsonl object"""
 35    all_cols: set[str] = set([col for item in data for col in item.keys()])
 36    return {
 37        "data[0]": data[0],
 38        "len(data)": len(data),
 39        "columns": {
 40            col: {
 41                "types": list(
 42                    set([type(item[col]).__name__ for item in data if col in item])
 43                ),
 44                "len": len([item[col] for item in data if col in item]),
 45            }
 46            for col in all_cols
 47            if col != _FORMAT_KEY
 48        },
 49    }

 93def zanj_external_serialize(
 94    jser: _ZANJ_pre,
 95    data: Any,
 96    path: ObjectPath,
 97    item_type: ExternalItemType,
 98    _format: str,
 99) -> JSONitem:
100    """stores a numpy array or jsonl externally in a ZANJ object
101
102    # Parameters:
103     - `jser: ZANJ`
104     - `data: Any`
105     - `path: ObjectPath`
106     - `item_type: ExternalItemType`
107
108    # Returns:
109     - `JSONitem`
110       json data with reference
111
112    # Modifies:
113     - modifies `jser._externals`
114    """
115    # get the path, make sure its unique
116    assert isinstance(path, tuple), (
117        f"path must be a tuple, got {type(path) = } {path = }"
118    )
119    joined_path: str = "/".join([str(p) for p in path])
120    archive_path: str = f"{joined_path}.{item_type}"
121
122    if archive_path in jser._externals:
123        raise ValueError(f"external path {archive_path} already exists!")
124    if any([p.startswith(joined_path) for p in jser._externals.keys()]):
125        raise ValueError(f"external path {joined_path} is a prefix of another path!")
126
127    # process the data if needed, assemble metadata
128    data_new: Any = data
129    output: dict = {
130        _FORMAT_KEY: _format,
131        _REF_KEY: archive_path,
132    }
133    if item_type == "npy":
134        # check type
135        data_type_str: str = str(type(data))
136        if data_type_str == "<class 'torch.Tensor'>":
137            # detach and convert
138            data_new = data.detach().cpu().numpy()
139        elif data_type_str == "<class 'numpy.ndarray'>":
140            pass
141        else:
142            # if not a numpy array, except
143            raise TypeError(f"expected numpy.ndarray, got {data_type_str}")
144        # get metadata
145        output.update(arr_metadata(data))
146    elif item_type.startswith("jsonl"):
147        # check via mro to avoid importing pandas
148        if any("pandas.core.frame.DataFrame" in str(t) for t in data.__class__.__mro__):
149            output["columns"] = data.columns.tolist()
150            data_new = data.to_dict(orient="records")
151        elif isinstance(data, (list, tuple, Iterable, Sequence)):
152            data_new = [
153                jser.json_serialize(item, tuple(path) + (i,))
154                for i, item in enumerate(data)
155            ]
156        else:
157            raise TypeError(
158                f"expected list or pandas.DataFrame for jsonl, got {type(data)}"
159            )
160
161        if all([isinstance(item, dict) for item in data_new]):
162            output.update(jsonl_metadata(data_new))
163
164    # store the item for external serialization
165    jser._externals[archive_path] = ExternalItem(
166        item_type=item_type,
167        data=data_new,
168        path=path,
169    )
170
171    return output
